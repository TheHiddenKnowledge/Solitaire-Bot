195 def __reset_game(self):
196 # Resetting GUI variables
197 self.moves = 0
198 self.__score = 0
199 self.__time = 0
200 self.__win = False
201 self.__clear_selected_cards()
202 # Resetting tableau index array
203 for a in range(len(self.tableau)):
204 for b in range(len(self.tableau[a])):
205 self.tableau[a][b] = -1
206 # Shuffling cards
207 random_idxs = rnd.sample(range(0, 52), 52)
208 # Assigning card indices to tableau
209 temp_idx = 0
210 for a in range(len(self.tableau)):
211 for b in range(a + 1):
212 idx = random_idxs[temp_idx]
213 self.tableau[a][b] = idx
214 # Only revealing card at the bottom of the tableau column
215 if b == a:
216 self.cards[idx].flipped = False
217 else:
218 self.cards[idx].flipped = True
219 temp_idx += 1
220 # Resetting stock index and array
221 self.stock_idx = -1
222 self.stock = []
223 # Assigning card indices to stock
224 for a in range(28, 52):
225 idx = random_idxs[a]
226 self.cards[idx].flipped = True
227 self.stock.append(idx)
228 # Resetting foundation piles
229 for a in range(len(self.found_idxs)):
230 for b in range(len(self.found_idxs[a])):
231 self.found_idxs[a][b] = -1

553 def __get_clicked(self, cursor):
554 # Click on revealed stock pile
555 if self.__stock_rects[0].collidepoint(cursor):
556 return ['stock_reveal', 0]
557 # Click on hidden stock pile
558 if self.__stock_rects[1].collidepoint(cursor):
559 return ['stock_hidden', 0]
560 # Click on foundation pile
561 for a in range(len(self.__found_rects)):
562 if self.__found_rects[a].collidepoint(cursor):
563 return ['foundation', a]
564 # Click on tableau card(s)
565 for a in range(len(self.tableau)):
566 for b in range(len(self.tableau[a])):
567 # Click checking rectangle
568 check_rect = pygame.Rect(0, 0, 0, 0)
569 # Current card index
570 idx = self.tableau[a][b]
571 # If not at the end of the tableau column
572 if b != len(self.tableau[a]) - 1:
573 # Next card index
574 next_idx = self.tableau[a][b + 1]
575 # If the next card is not empty
576 if next_idx >= 0:
577 card_rect = self.cards[idx].rect
578 next_card_rect = self.cards[next_idx].rect
579 check_rect = copy.copy(card_rect)
580 check_rect.height = next_card_rect.y - card_rect.y
581 # If the next card is empty
582 else:
583 check_rect = self.cards[idx].rect
584 # End of the tableau column
585 else:
586 if idx >= 0:
587 check_rect = self.cards[idx].rect
588 # Only selects the card if it is not empty and it is clicked
589 if idx >= 0 and not self.cards[idx].flipped:
590 if check_rect.collidepoint(cursor):
591 return ['tableau_card', self.tableau[a][b]]
592 # Click on tableau pile
593 for a in range(len(self.__tableau_rects)):
594 if self.__tableau_rects[a].collidepoint(cursor):
595 return ['tableau_pile', a]
596 # Return value if nothing was clicked on
597 return ['none', 0]

693 def run_game(self):
694 for event in pygame.event.get():
695 if event.type == pygame.QUIT:
696 self.quit = True
697 if event.type == pygame.MOUSEBUTTONDOWN:
698 if event.button == 1:
699 if self.__reset_rect.collidepoint(event.pos):
700 self.__reset_game()
701 if not self.__win:
702 if self.__click_handler(event.pos):
703 self.moves += 1
704 self.__get_game_win()
705 if event.type == pygame.KEYDOWN:
706 if event.key == pygame.K_SPACE:
707 if not self.__win:
708 self.src_entity = ['stock_hidden', 0]
709 self.dest_entity = ['none', 0]
710 self.__increment_stock()
711 self.moves += 1
712 self.__screen.fill(self.__screen_color)
713 self.__get_card_positions()
714 self.__draw_game()
715 self.__draw_gui()
716 self.__time += self.__clock.get_time() / 1000
717 self.__clock.tick(60)
718 pygame.display.flip()
