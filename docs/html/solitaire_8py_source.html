194 def __reset_game(self):
195 # Resetting GUI variables
196 self.__moves = 0
197 self.__score = 0
198 self.__time = 0
199 self.__win = False
200 self.__clear_selected_cards()
201 # Resetting tableau index array
202 for a in range(len(self.__tableau)):
203 for b in range(len(self.__tableau[a])):
204 self.__tableau[a][b] = -1
205 # Shuffling cards
206 random_idxs = rnd.sample(range(0, 52), 52)
207 # Assigning card indices to tableau
208 temp_idx = 0
209 for a in range(len(self.__tableau)):
210 for b in range(a + 1):
211 idx = random_idxs[temp_idx]
212 self.__tableau[a][b] = idx
213 # Only revealing card at the bottom of the tableau column
214 if b == a:
215 self.__cards[idx].flipped = False
216 else:
217 self.__cards[idx].flipped = True
218 temp_idx += 1
219 # Resetting stock index and array
220 self.__stock_idx = -1
221 self.__stock = []
222 # Assigning card indices to stock
223 for a in range(28, 52):
224 idx = random_idxs[a]
225 self.__cards[idx].flipped = True
226 self.__stock.append(idx)
227 # Resetting foundation piles
228 for a in range(len(self.__found_idxs)):
229 for b in range(len(self.__found_idxs[a])):
230 self.__found_idxs[a][b] = -1
231 self.__found_suits[a] = -1
232 self.__found_ranks[a] = -1

393 def __get_clicked(self, cursor):
394 # Click on revealed stock pile
395 if self.__stock_rects[0].collidepoint(cursor):
396 return ['stock_reveal', 0, 0]
397 # Click on hidden stock pile
398 if self.__stock_rects[1].collidepoint(cursor):
399 return ['stock_hidden', 0, 0]
400 # Click on foundation pile
401 for a in range(len(self.__found_rects)):
402 if self.__found_rects[a].collidepoint(cursor):
403 return ['foundation', a, 0]
404 # Click on tableau card(s)
405 for a in range(len(self.__tableau)):
406 for b in range(len(self.__tableau[a])):
407 # Click checking rectangle
408 check_rect = pygame.Rect(0, 0, 0, 0)
409 # Current card index
410 idx = self.__tableau[a][b]
411 # If not at the end of the tableau column
412 if b != len(self.__tableau[a]) - 1:
413 # Next card index
414 next_idx = self.__tableau[a][b + 1]
415 # If the next card is not empty
416 if next_idx >= 0:
417 card_rect = self.__cards[idx].rect
418 next_card_rect = self.__cards[next_idx].rect
419 check_rect = copy.copy(card_rect)
420 check_rect.height = next_card_rect.y - card_rect.y
421 # If the next card is empty
422 else:
423 check_rect = self.__cards[idx].rect
424 # End of the tableau column
425 else:
426 if idx >= 0:
427 check_rect = self.__cards[idx].rect
428 # Only selects the card if it is not empty and it is clicked
429 if idx >= 0 and not self.__cards[idx].flipped:
430 if check_rect.collidepoint(cursor):
431 return ['tableau_card', a, b]
432 # Click on tableau pile
433 for a in range(len(self.__tableau_rects)):
434 if self.__tableau_rects[a].collidepoint(cursor):
435 return ['tableau_pile', a, -1]
436 # Return value if nothing was clicked on
437 return ['none', 0, 0]

442 def __can_card_move(self, clicked_entity):
443 select_col = self.__selected_card[1]
444 select_row = self.__selected_card[2]
445 clicked_col = clicked_entity[1]
446 clicked_row = clicked_entity[2]
447 if clicked_entity[0] == 'foundation':
448 # Selected card
449 card = None
450 if self.__selected_card[0] == 'tableau_card':
451 card_idx = self.__tableau[select_col][
452 select_row]
453 # Don't move card to foundation if it has other cards attached
454 if select_row != len(self.__tableau[0]) - 1:
455 next_card_idx = self.__tableau[select_col][
456 select_row + 1]
457 if next_card_idx >= 0:
458 return False
459 card = self.__cards[card_idx]
460 elif self.__selected_card[0] == 'stock_reveal':
461 card = self.__cards[self.__stock[self.__stock_idx]]
462 if card is None:
463 return False
464 found_idx = clicked_col
465 # Only move card to foundation if the suit is empty or matches
466 if (self.__found_ranks[found_idx] < 0
467 or self.__found_suits[found_idx] == card.suit):
468 if (card.rank - self.__found_ranks[found_idx]) == 1:
469 return True
470 elif clicked_entity[0] == 'tableau_card':
471 # Selected card
472 select_card = None
473 if self.__selected_card[0] == 'tableau_card':
474 select_card = self.__cards[
475 self.__tableau[select_col]
476 [select_row]]
477 elif self.__selected_card[0] == 'stock_reveal':
478 select_card = self.__cards[self.__stock[self.__stock_idx]]
479 elif self.__selected_card[0] == 'foundation':
480 rank = self.__found_ranks[select_col]
481 card_idx = self.__found_idxs[select_col][rank]
482 select_card = self.__cards[card_idx]
483 if select_card is None:
484 return False
485 # Destination card
486 dest_card = self.__cards[
487 self.__tableau[clicked_col][clicked_row]]
488 # Checking if suits are opposite colors
489 if ((select_card.suit < 2 and dest_card.suit > 1)
490 or (select_card.suit > 1 and dest_card.suit < 2)):
491 # Checking if rank is in order
492 if (dest_card.rank - select_card.rank) == 1:
493 return True
494 elif clicked_entity[0] == 'tableau_pile':
495 # Selected card
496 select_card = None
497 if self.__selected_card[0] == 'tableau_card':
498 select_card = self.__cards[
499 self.__tableau[select_col]
500 [select_row]]
501 elif self.__selected_card[0] == 'stock_reveal':
502 select_card = self.__cards[self.__stock[self.__stock_idx]]
503 elif self.__selected_card[0] == 'foundation':
504 rank = self.__found_ranks[select_col]
505 card_idx = self.__found_idxs[select_col][rank]
506 select_card = self.__cards[card_idx]
507 if select_card is None:
508 return False
509 # Card can only move to the tableau pile start if it is a king
510 # and the pile is empty
511 if select_card.rank == 12:
512 if self.__tableau[clicked_col][0] == -1:
513 return True
514 return False

519 def __click_handler(self, cursor):
520 clicked_entity = self.__get_clicked(cursor)
521 select_col = self.__selected_card[1]
522 select_row = self.__selected_card[2]
523 clicked_col = clicked_entity[1]
524 clicked_row = clicked_entity[2]
525 if clicked_entity[0] == 'tableau_card':
526 # If a card was not selected previously or the current selection
527 # is in the same tableau column as the previous selection
528 if (self.__selected_card[0] == 'none'
529 or (self.__selected_card[0] == 'tableau_card'
530 and select_col == clicked_col)):
531 # Selects the card(s)
532 self.__clear_selected_cards()
533 self.__selected_card = clicked_entity
534 for a in range(clicked_row, len(self.__tableau[clicked_col])):
535 idx = self.__tableau[clicked_col][a]
536 if idx >= 0:
537 self.__cards[idx].selected = True
538 return False
539 else:
540 if self.__can_card_move(clicked_entity):
541 # If the previously selected card was in the tableau
542 if self.__selected_card[0] == 'tableau_card':
543 self.__move_tableau_cards(select_col, select_row,
544 clicked_col, clicked_row)
545 self.__flip_next_card(select_col, select_row)
546 # If the previously selected card was in the stock
547 elif self.__selected_card[0] == 'stock_reveal':
548 # Adds stock card to the clicked cards column
549 self.__tableau[clicked_col][clicked_row + 1] = (
550 self.__stock)[self.__stock_idx]
551 # Removes card index from the stock
552 self.__stock.pop(self.__stock_idx)
553 self.__stock_idx -= 1
554 # If the previously selected card was in the foundation
555 elif self.__selected_card[0] == 'foundation':
556 # Adds foundation card to the clicked cards column
557 rank = self.__found_ranks[select_col]
558 card_idx = self.__found_idxs[select_col][rank]
559 self.__tableau[clicked_col][clicked_row + 1] = card_idx
560 # Removes card from foundation pile
561 self.__found_idxs[select_col][rank] = -1
562 self.__found_ranks[select_col] -= 1
563 # Resets foundation suit if empty
564 if self.__found_ranks[select_col] < 0:
565 self.__found_suits[select_col] = -1
566 else:
567 return False
568 self.__clear_selected_cards()
569 return True
570 else:
571 return False
572 elif clicked_entity[0] == 'tableau_pile':
573 if self.__can_card_move(clicked_entity):
574 # If the previously selected card was in the tableau
575 if self.__selected_card[0] == 'tableau_card':
576 self.__move_tableau_cards(select_col, select_row,
577 clicked_col, clicked_row)
578 self.__flip_next_card(select_col, select_row)
579 # If the previously selected card was in the stock
580 elif self.__selected_card[0] == 'stock_reveal':
581 # Adds stock card to the clicked pile column
582 self.__tableau[clicked_col][0] = (
583 self.__stock)[self.__stock_idx]
584 # Removing card index from the stock
585 self.__stock.pop(self.__stock_idx)
586 self.__stock_idx -= 1
587 else:
588 return False
589 self.__clear_selected_cards()
590 return True
591 else:
592 return False
593 elif clicked_entity[0] == 'stock_reveal':
594 if len(self.__stock) > 0 and self.__stock_idx >= 0:
595 # Selecting the revealed card in the stock
596 self.__clear_selected_cards()
597 self.__selected_card = clicked_entity
598 card = self.__cards[self.__stock[self.__stock_idx]]
599 card.selected = True
600 return False
601 else:
602 return False
603 elif clicked_entity[0] == 'stock_hidden':
604 self.__increment_stock()
605 return True
606 elif clicked_entity[0] == 'foundation':
607 # If a card was not selected previously and the foundation pile
608 # is not empty
609 if (self.__selected_card[0] == 'none'
610 and self.__found_suits[clicked_col] >= 0):
611 self.__clear_selected_cards()
612 rank = self.__found_ranks[clicked_col]
613 card = self.__cards[self.__found_idxs[clicked_col][rank]]
614 self.__selected_card = clicked_entity
615 card.selected = True
616 return False
617 # If a card was selected previously
618 else:
619 if self.__can_card_move(clicked_entity):
620 if self.__selected_card[0] == 'tableau_card':
621 card_idx = self.__tableau[select_col][select_row]
622 # Adding card to the foundation
623 if self.__found_suits[clicked_col] == -1:
624 self.__found_suits[clicked_col] = (
625 self.__cards[card_idx].suit)
626 self.__found_ranks[clicked_col] += 1
627 rank = self.__found_ranks[clicked_col]
628 self.__found_idxs[clicked_col][rank] = card_idx
629 # Removing card from the tableau
630 self.__tableau[select_col][select_row] = -1
631 self.__flip_next_card(select_col, select_row)
632 self.__clear_selected_cards()
633 elif self.__selected_card[0] == 'stock_reveal':
634 card_idx = self.__stock[self.__stock_idx]
635 # Adding card to the foundation
636 if self.__found_suits[clicked_col] == -1:
637 self.__found_suits[clicked_col] = (
638 self.__cards[card_idx].suit)
639 self.__found_ranks[clicked_col] += 1
640 rank = self.__found_ranks[clicked_col]
641 self.__found_idxs[clicked_col][rank] = card_idx
642 # Removing card index from the stock
643 self.__stock.pop(self.__stock_idx)
644 self.__stock_idx -= 1
645 self.__clear_selected_cards()
646 return True
647 else:
648 return False
649 else:
650 self.__clear_selected_cards()
651 return False

656 win_check = 1

667 def run_game(self):
668 for event in pygame.event.get():
669 if event.type == pygame.QUIT:
670 self.quit = True
671 if event.type == pygame.MOUSEBUTTONDOWN:
672 if event.button == 1:
673 if self.__reset_rect.collidepoint(event.pos):
674 self.__reset_game()
675 if not self.__win:
676 if self.__click_handler(event.pos):
677 self.__moves += 1
678 self.__get_game_win()
679 if event.type == pygame.KEYDOWN:
680 if event.key == pygame.K_SPACE:
681 if not self.__win:
682 self.__increment_stock()
683 self.__moves += 1
684 self.__screen.fill(self.__screen_color)
685 self.__get_card_positions()
686 self.__draw_game()
687 self.__draw_gui()
688 self.__time += self.__clock.get_time() / 1000
689 self.__clock.tick(60)
690 pygame.display.flip()
