65 def use_active_functions(self):
66 if self.activation == 'input':
67 self.active_values = self.original_values.copy()
68 elif self.activation == 'sigmoid':
69 self.active_values = 1 / (1 + np.exp(-self.original_values))
70 elif self.activation == 'tanh':
71 self.active_values = np.tanh(self.original_values)
72 elif self.activation == 'relu':
73 self.active_values = np.where(self.original_values > 0,
74 self.original_values,
75 .1 * self.original_values)
76 else:
77 raise NetError(self.activation, "Invalid activation function.")

81 def get_active_derivatives(self):
82 if self.activation == 'input':
83 self.active_derivatives = np.zeros_like(self.original_values)
84 elif self.activation == 'sigmoid':
85 self.active_derivatives = (self.active_values
86 * (1 - self.active_values))
87 elif self.activation == 'tanh':
88 self.active_derivatives = 1 - np.power(self.active_values, 2)
89 elif self.activation == 'relu':
90 self.active_derivatives = np.where(self.active_values > 0, 1, .1)
91 else:
92 raise NetError(self.activation, "Invalid activation function.")

215 def forward_pass(self, net_input):
216 self.__net_layers[0].set_layer_values(net_input)
217 for a in range(len(self.__weights)):
218 current_values = self.__net_layers[a].active_values
219 # Transforming the previous layer into the next layer
220 next_values = (np.matmul(self.__weights[a], current_values)
221 + self.__biases[a])
222 self.__net_layers[a + 1].set_layer_values(next_values)
223 # Returns the last layer (output)
224 return self.__net_layers[-1].active_values

229 def __get_gradients(self, expected_output):
230 # List of layer derivatives with respect to cost
231 # (used for backpropagation)
232 layer_derivatives = []
233 for net_layer in self.__net_layers:
234 layer_derivatives.append(np.zeros_like(net_layer.original_values))
235 for a in reversed(range(len(self.__weights))):
236 active_values = self.__net_layers[a].active_values
237 next_active_values = self.__net_layers[a + 1].active_values
238 next_active_derivatives = self.__net_layers[a + 1].active_derivatives
239 for b in range(self.__weights[a].shape[0]):
240 # Getting bias gradient for last hidden layer
241 if a == len(self.__weights) - 1:
242 self.__bias_gradient[a][b] = (
243 next_active_derivatives[b][0]
244 * 2 * (next_active_values[b][0]
245 - expected_output[b][0])
246 )
247 # Getting bias gradient used backpropagation
248 else:
249 self.__bias_gradient[a][b] = (
250 next_active_derivatives[b][0]
251 * layer_derivatives[a + 1][b][0]
252 )
253 for c in range(self.__weights[a].shape[1]):
254 if a == len(self.__weights) - 1:
255 # Getting weight gradient for last hidden layer
256 self.__weight_gradient[a][b][c] = (
257 active_values[c][0]
258 * next_active_derivatives[b][0]
259 * 2 * (next_active_values[b][0]
260 - expected_output[b][0])
261 )
262 # Calculating layer derivatives with respect to cost
263 layer_derivatives[a][c][0] += (
264 self.__weights[a][b][c]
265 * next_active_derivatives[b][0]
266 * 2 * (next_active_values[b][0]
267 - expected_output[b][0])
268 )
269 else:
270 # Getting weight gradient using backpropagation
271 self.__weight_gradient[a][b][c] = (
272 active_values[c][0]
273 * next_active_derivatives[b][0]
274 * layer_derivatives[a + 1][b][0]
275 )
276 # Calculating layer derivatives with respect to cost
277 layer_derivatives[a][c][0] += (
278 self.__weights[a][b][c]
279 * next_active_derivatives[b][0]
280 * layer_derivatives[a + 1][b][0]

313 def __step_mini_batch(self, mini_batch):
314 avg_weight_gradient = copy.deepcopy(self.__weights)
315 avg_bias_gradient = copy.deepcopy(self.__biases)
316 for a in range(len(self.__weights)):
317 avg_weight_gradient[a].fill(0)
318 avg_bias_gradient[a].fill(0)
319 avg_cost = 0
320 for a in range(len(mini_batch)):
321 self.forward_pass(mini_batch[a][0])
322 self.__get_gradients(mini_batch[a][1])
323 for b in range(len(self.__weights)):
324 avg_weight_gradient[b] += (self.__weight_gradient[b]
325 / len(mini_batch))
326 avg_bias_gradient[b] += (self.__bias_gradient[b]
327 / len(mini_batch))
328 cost = 0
329 for b in range(len(mini_batch[a][1])):
330 output_values = self.__net_layers[-1].active_values
331 cost += (output_values[b][0] - mini_batch[a][1][b][0]) ** 2
332 avg_cost += cost / len(mini_batch)
333 # Descending the gradient using the momentum theorem for neural nets
334 for a in range(len(self.__weights)):
335 weight_velocity = (self.__beta * self.__prev_weight_velocity[a]
336 + (1 - self.__beta) * avg_weight_gradient[a])
337 bias_veloctiy = (self.__beta * self.__prev_bias_velocity[a]
338 + (1 - self.__beta) * avg_bias_gradient[a])
339 self.__weights[a] -= self.__rate * weight_velocity
340 self.__biases[a] -= self.__rate * bias_veloctiy
341 np.copyto(self.__prev_weight_velocity[a], weight_velocity)
342 np.copyto(self.__prev_bias_velocity[a], bias_veloctiy)
343 return avg_cost

348 def step_epoch(self, batch):
349 avg_cost = 0
350 for a in range(self.__mini_batch_max_index):
351 start_idx = a * self.__mini_batch_size
352 end_idx = (a + 1) * self.__mini_batch_size
353 mini_batch = batch[start_idx:end_idx]
354 avg_cost = self.__step_mini_batch(mini_batch)
355 self.__epoch += 1
